        "import requests\n",
        "\n",
        "BASE_URL = \"https://dummyjson.com/products\"\n",
        "\n",
        "\n",
        "def fetch_products(limit=20):\n",
        "    url = f\"{BASE_URL}?limit={limit}\"\n",
        "    r = requests.get(url)\n",
        "    r.raise_for_status()\n",
        "    return r.json()[\"products\"]\n",
        "\n",
        "\n",
        "def print_products_table(products):\n",
        "    print(f\"{'ID':<4}| {'Title':<32}| {'Category':<15}| {'Price':<9}| Rating\")\n",
        "    print(\"-\" * 80)\n",
        "    for p in products:\n",
        "        print(\n",
        "            f\"{p['id']:<4}| \"\n",
        "            f\"{p['title']:<32}| \"\n",
        "            f\"{p['category']:<15}| \"\n",
        "            f\"${p['price']:<8.2f}| {p['rating']}\"\n",
        "        )\n",
        "\n",
        "\n",
        "def filter_and_sort_by_rating(products, min_rating=4.5):\n",
        "    result = [p for p in products if p[\"rating\"] >= min_rating]\n",
        "    return sorted(result, key=lambda x: x[\"price\"], reverse=True)\n",
        "\n",
        "\n",
        "def fetch_by_category(category):\n",
        "    url = f\"{BASE_URL}/category/{category}\"\n",
        "    r = requests.get(url)\n",
        "    r.raise_for_status()\n",
        "    return r.json()[\"products\"]\n",
        "\n",
        "\n",
        "def create_product(payload):\n",
        "    url = f\"{BASE_URL}/add\"\n",
        "    r = requests.post(url, json=payload)\n",
        "    r.raise_for_status()\n",
        "    return r.json()\n",
        "\n",
        "\n",
        "# ---------------------- Step 1 ----------------------\n",
        "print(\"Fetching 20 products...\\n\")\n",
        "all_products = fetch_products(20)\n",
        "print_products_table(all_products)\n",
        "\n",
        "# ---------------------- Step 2 ----------------------\n",
        "print(\"\\nFiltered Products (Rating >= 4.5):\\n\")\n",
        "high_rated = filter_and_sort_by_rating(all_products, 4.5)\n",
        "for p in high_rated:\n",
        "    print(f\"{p['title']} | Price: ${p['price']:.2f} | Rating: {p['rating']}\")\n",
        "\n",
        "# ---------------------- Step 3 ----------------------\n",
        "print(\"\\nLaptops Category:\\n\")\n",
        "laptops = fetch_by_category(\"laptops\")\n",
        "for lp in laptops:\n",
        "    print(f\"{lp['title']} | Price: ${lp['price']:.2f}\")\n",
        "\n",
        "# ---------------------- Step 4 ----------------------\n",
        "print(\"\\nSending POST Request...\\n\")\n",
        "payload = {\n",
        "    \"title\": \"My Custom Product\",\n",
        "    \"price\": 999,\n",
        "    \"category\": \"electronics\",\n",
        "    \"description\": \"A product I created via API\",\n",
        "}\n",
        "response_data = create_product(payload)\n",
        "\n",
        "print(\"Response from server:\")\n",
        "print(response_data)\n"
